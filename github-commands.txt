# GitHub Commands for Workflow 1 (Run these after creating empty GitHub repo)

# 1. Add remote origin (replace URL with your actual repository URL)
git remote add origin https://github.com/YOUR_USERNAME/ProjectStudent.git

# 2. Rename master to main (GitHub default)
git branch -M main

# 3. Push all branches to GitHub
git push -u origin main
git push -u origin staging  
git push -u origin feature/initial-setup

# 4. Delete feature branch locally and remotely
git branch -d feature/initial-setup
git push origin --delete feature/initial-setup

# Verify branches on GitHub after pushing
